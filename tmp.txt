class _ScheduleRow extends StatelessWidget {
  const _ScheduleRow({
    required this.item,
    required this.timeFormat,
    required this.groupLabel,
    required this.isLast,
    required this.busy,
    required this.onToggle,
    required this.onOpenDetails,
  });

  final sched.ClassItem item;
  final DateFormat timeFormat;
  final String groupLabel;
  final bool isLast;
  final bool busy;
  final ValueChanged<bool> onToggle;
  final VoidCallback onOpenDetails;

  @override
  Widget build(BuildContext context) {
    final theme = Theme.of(context);
    final colors = theme.colorScheme;
    final timeLabel =
        '${_formatTime(item.start)} - ${_formatTime(item.end)}';
    final location = item.room?.trim() ?? '';
    final instructor = item.instructor?.trim() ?? '';
    final disabled = !item.enabled;
    final start = _timeFromMinutes(_minutesFromText(item.start));
    final dayAbbrev = groupLabel.isNotEmpty
        ? groupLabel.substring(0, math.min(3, groupLabel.length)).toUpperCase()
        : DateFormat('EEE').format(start).toUpperCase();

    return InkWell(
      onTap: onOpenDetails,
      borderRadius: AppTokens.radius.lg,
      child: Padding(
        padding: EdgeInsets.only(
          bottom: isLast ? 0 : 12,
          top: 0,
        ),
        child: Container(
          padding: const EdgeInsets.fromLTRB(18, 16, 12, 16),
          decoration: BoxDecoration(
            color: disabled
                ? colors.surfaceContainerHigh.withValues(alpha: 0.65)
                : colors.surfaceContainerHigh,
            borderRadius: AppTokens.radius.lg,
            border: Border.all(
              color: disabled
                  ? colors.outlineVariant.withValues(alpha: 0.2)
                  : colors.outlineVariant.withValues(alpha: 0.3),
            ),
          ],
          child: Row(
            crossAxisAlignment: CrossAxisAlignment.start,
            children: [
              Column(
                children: [
                  Container(
                    padding:
                        const EdgeInsets.symmetric(horizontal: 12, vertical: 8),
                    decoration: BoxDecoration(
                      color: colors.primary.withValues(alpha: 0.12),
                      borderRadius: BorderRadius.circular(16),
                    ),
                    child: Text(
                      DateFormat('h:mm a').format(start),
                      style: theme.textTheme.labelMedium?.copyWith(
                        fontWeight: FontWeight.w700,
                        color: colors.primary,
                      ),
                    ),
                  ),
                  const SizedBox(height: 6),
                  Text(
                    dayAbbrev,
                    style: theme.textTheme.labelSmall?.copyWith(
                      fontWeight: FontWeight.w600,
                      color: colors.onSurfaceVariant.withValues(alpha: 0.7),
                    ),
                  ),
                ],
              ),
              const SizedBox(width: 14),
              Expanded(
                child: Column(
                  crossAxisAlignment: CrossAxisAlignment.start,
                  children: [
                    Row(
                      crossAxisAlignment: CrossAxisAlignment.start,
                      children: [
                        Expanded(
                          child: Text(
                            item.title ?? item.code ?? 'Class',
                            style: theme.textTheme.titleMedium?.copyWith(
                              fontWeight: FontWeight.w600,
                            ),
                          ),
                        ),
                        const SizedBox(width: 8),
                        Switch.adaptive(
                          value: item.enabled,
                          onChanged: busy ? null : onToggle,
                        ),
                      ],
                    ),
                    const SizedBox(height: 6),
                    _ScheduleInfoRow(
                      icon: Icons.schedule_rounded,
                      label: timeLabel,
                    ),
                    if (location.isNotEmpty) ...[
                      const SizedBox(height: 4),
                      _ScheduleInfoRow(
                        icon: Icons.location_on_outlined,
                        label: location,
                      ),
                    ],
                    if (instructor.isNotEmpty) ...[
                      const SizedBox(height: 4),
                      _ScheduleInfoRow(
                        icon: Icons.person_outline,
                        label: instructor,
                      ),
                    ],
                    if (item.isCustom) ...[
                      const SizedBox(height: 8),
                      Text(
                        'Custom class',
                        style: theme.textTheme.bodySmall?.copyWith(
                          color: colors.primary,
                          fontWeight: FontWeight.w600,
                        ),
                      ),
                    ],
                    if (disabled) ...[
                      const SizedBox(height: 6),
                      Text(
                        'Class disabled',
                        style: theme.textTheme.bodySmall?.copyWith(
                          color: colors.error,
                          fontWeight: FontWeight.w600,
                        ),
                      ),
                    ],
                  ],
                ),
              ),
            ],
          ),
        ),
      ),
    );
  }
}
